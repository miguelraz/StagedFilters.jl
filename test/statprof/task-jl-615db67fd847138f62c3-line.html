28 """
29 CompositeException
30
31 Wrap a `Vector` of exceptions thrown by a [`Task`](@ref) (e.g. generated from a remote worker over a channel
32 or an asynchronously executing local I/O write or a remote worker under `pmap`) with information about the series of exceptions.
33 For example, if a group of workers are executing several tasks, and multiple workers fail, the resulting `CompositeException` will
34 contain a "bundle" of information from each worker indicating where and why the exception(s) occurred.
35 """
36 struct CompositeException <: Exception
37 exceptions::Vector{Any}
38 CompositeException() = new(Any[])
39 CompositeException(exceptions) = new(exceptions)
40 end
41 length(c::CompositeException) = length(c.exceptions)
42 push!(c::CompositeException, ex) = push!(c.exceptions, ex)
43 isempty(c::CompositeException) = isempty(c.exceptions)
44 iterate(c::CompositeException, state...) = iterate(c.exceptions, state...)
45 eltype(::Type{CompositeException}) = Any

47 function showerror(io::IO, ex::CompositeException)
48 if !isempty(ex)
49 showerror(io, ex.exceptions[1])
50 remaining = length(ex) - 1
51 if remaining > 0
52 print(io, string("\n\n...and ", remaining, " more exception(s).\n"))
53 end
54 else
55 print(io, "CompositeException()\n")
56 end

188 """
189 task_local_storage(body, key, value)
190
191 Call the function `body` with a modified task-local storage, in which `value` is assigned to
192 `key`; the previous value of `key`, or lack thereof, is restored afterwards. Useful
193 for emulating dynamic scoping.
194 """
195 function task_local_storage(body::Function, key, val)
196 tls = task_local_storage()
197 hadkey = haskey(tls, key)
198 old = get(tls, key, nothing)
199 tls[key] = val
200 try
201 return body()
202 finally
203 hadkey ? (tls[key] = old) : delete!(tls, key)
204 end

207 # NOTE: you can only wait for scheduled tasks
208 function wait(t::Task)
209 t === current_task() && error("deadlock detected: cannot wait on current task")
210 if !istaskdone(t)
211 lock(t.donenotify)
212 try
213 while !istaskdone(t)
214 wait(t.donenotify)
215 end
216 finally
217 unlock(t.donenotify)
218 end
219 end
220 if istaskfailed(t)
221 throw(t.exception)
222 end
223 nothing
224 end

242 function sync_end(refs)
243 c_ex = CompositeException()
244 for r in refs
245 try
246 wait(r)
247 catch
248 if !isa(r, Task) || (isa(r, Task) && !istaskfailed(r))
249 rethrow()
250 end
251 finally
252 if isa(r, Task) && istaskfailed(r)
253 push!(c_ex, CapturedException(task_result(r), r.backtrace))
254 end
255 end
256 end
257
258 if !isempty(c_ex)
259 throw(c_ex)
260 end
261 nothing
262 end

311 # runtime system hook called when a task finishes
312 function task_done_hook(t::Task)
313 # `finish_task` sets `sigatomic` before entering this function
314 err = istaskfailed(t)
315 result = task_result(t)
316 handled = false
317 if err
318 t.backtrace = catch_backtrace()
319 end
320
321 donenotify = t.donenotify
322 if isa(donenotify, ThreadSynchronizer)
323 lock(donenotify)
324 try
325 if !isempty(donenotify.waitq)
326 handled = true
327 notify(donenotify, result, true, err)
328 end
329 finally
330 unlock(donenotify)
331 end
332 end
333
334 # Execute any other hooks registered in the TLS
335 if isa(t.storage, IdDict) && haskey(t.storage, :TASKDONE_HOOKS)
336 foreach(hook -> hook(t), t.storage[:TASKDONE_HOOKS])
337 delete!(t.storage, :TASKDONE_HOOKS)
338 handled = true
339 end
340
341 if err && !handled && Threads.threadid() == 1
342 if isa(result, InterruptException) && isdefined(Base, :active_repl_backend) &&
343 active_repl_backend.backend_task.state == :runnable && isempty(Workqueue) &&
344 active_repl_backend.in_eval
345 throwto(active_repl_backend.backend_task, result) # this terminates the task
346 end
347 end
348 # Clear sigatomic before waiting
349 sigatomic_end()
350 try
351 wait() # this will not return
352 catch e
353 # If an InterruptException happens while blocked in the event loop, try handing
354 # the exception to the REPL task since the current task is done.
355 # issue #19467
356 if Threads.threadid() == 1 &&
357 isa(e, InterruptException) && isdefined(Base, :active_repl_backend) &&
358 active_repl_backend.backend_task.state == :runnable && isempty(Workqueue) &&
359 active_repl_backend.in_eval
360 throwto(active_repl_backend.backend_task, e)
361 else
362 rethrow()
363 end
364 end
365 end

434 function enq_work(t::Task)
435 (t.state == :runnable && t.queue === nothing) || error("schedule: Task not runnable")
436 tid = Threads.threadid(t)
437 # Note there are three reasons a Task might be put into a sticky queue
438 # even if t.sticky == false:
439 # 1. The Task's stack is currently being used by the scheduler for a certain thread.
440 # 2. There is only 1 thread.
441 # 3. The multiq is full (can be fixed by making it growable).
442 if t.sticky || tid != 0 || Threads.nthreads() == 1
443 if tid == 0
444 tid = Threads.threadid()
445 ccall(:jl_set_task_tid, Cvoid, (Any, Cint), t, tid-1)
446 end
447 push!(Workqueues[tid], t)
448 else
449 tid = 0
450 if ccall(:jl_enqueue_task, Cint, (Any,), t) != 0
451 # if multiq is full, give to a random thread (TODO fix)
452 tid = mod(time_ns() % Int, Threads.nthreads()) + 1
453 ccall(:jl_set_task_tid, Cvoid, (Any, Cint), t, tid-1)
454 push!(Workqueues[tid], t)
455 end
456 end
457 ccall(:jl_wakeup_thread, Cvoid, (Int16,), (tid - 1) % Int16)
458 return t
459 end

463 """
464 schedule(t::Task, [val]; error=false)
465
466 Add a [`Task`](@ref) to the scheduler's queue. This causes the task to run constantly when the system
467 is otherwise idle, unless the task performs a blocking operation such as [`wait`](@ref).
468
469 If a second argument `val` is provided, it will be passed to the task (via the return value of
470 [`yieldto`](@ref)) when it runs again. If `error` is `true`, the value is raised as an exception in
471 the woken task.
472
473 # Examples
474 ```jldoctest
475 julia> a5() = sum(i for i in 1:1000);
476
477 julia> b = Task(a5);
478
479 julia> istaskstarted(b)
480 false
481
482 julia> schedule(b);
483
484 julia> yield();
485
486 julia> istaskstarted(b)
487 true
488
489 julia> istaskdone(b)
490 true
491 ```
492 """
493 function schedule(t::Task, @nospecialize(arg); error=false)
494 # schedule a task to be (re)started with the given value or exception
495 t.state == :runnable || Base.error("schedule: Task not runnable")
496 if error
497 t.queue === nothing || Base.list_deletefirst!(t.queue, t)
498 t.exception = arg
499 else
500 t.queue === nothing || Base.error("schedule: Task not runnable")
501 t.result = arg
502 end
503 enq_work(t)
504 return t
505 end

565 function ensure_rescheduled(othertask::Task)
566 ct = current_task()
567 W = Workqueues[Threads.threadid()]
568 if ct !== othertask && othertask.state == :runnable
569 # we failed to yield to othertask
570 # return it to the head of a queue to be retried later
571 tid = Threads.threadid(othertask)
572 Wother = tid == 0 ? W : Workqueues[tid]
573 pushfirst!(Wother, othertask)
574 end
575 # if the current task was queued,
576 # also need to return it to the runnable state
577 # before throwing an error
578 list_deletefirst!(W, ct)
579 nothing
580 end

582 function trypoptask(W::StickyWorkqueue)
583 isempty(W) && return
584 t = popfirst!(W)

586 # assume this somehow got queued twice,

607 W = Workqueues[Threads.threadid()]
608 reftask = poptaskref(W)
609 result = try_yieldto(ensure_rescheduled, reftask)
